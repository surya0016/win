1. Write a program to accept a number & display the factorial of the number
#Accpet a number and caluclate it's factorial 
while True: 
    n = int(input("Enter number : ")) 
    if n == 0 or n == 1: 
        print(f"The factorial of the given number {n} is 1") 
        break 
    elif n<0: 
        print(f"Factorial of a negative number cannot be found") 
    else: 
        fact = 1 
        i = 1 
        while (i<=n): 
            fact*=i 
            i+=1 
        print(f"The factorial of the given number {n} is {fact}") 
        break 



2. Write a program to accept a number and check if its palindrome or not
#Accept a number from user and check if it is a palindrome or not 
n = int(input("Enter number : ")) 
while n < 10: 
    print("Palindrome cannot be made out for a single digit number") 
    n = int(input("Enter number : ")) 
num = n 
rev = 0 
while n != 0: 
    digit = n % 10 
    rev = rev * 10 + digit 
    n = n // 10 
if num == rev: 
    print("The given number is a palindrome") 
else: 
    print("The given number is not a palindrome")
